# slice

* 切片虽然在作为入参传递时是传引用，但是这里的引用指的是切片底层数组的引用。切片本身仍然是复制后传递的。

  ```go
  package main
  
  import (
  	"fmt"
  )
  
  func foo1(nums []int) {
  	nums = append(nums, 2)
  }
  
  func foo2(nums []int) {
  	nums[0] = 2
  }
  
  func foo3(nums *[]int) {
  	// *&x == x
  	*nums = append(*nums, 3)
  }
  
  func main() {
  	nums := []int{1}
  	foo1(nums)
  	fmt.Println(nums)
  	foo2(nums)
  	fmt.Println(nums)
  	foo3(&nums)
  	fmt.Println(nums)
  }
  ```
